import torch
from transformers import AutoTokenizer
from realhf.api.core.data_api import SequenceSample


def pack_input_ids(input_ids: torch.Tensor, seqlen: torch.Tensor) -> torch.Tensor:
    """
    将 input_ids 按 seqlen 拼接成 packed_input_ids。
    Args:
        input_ids: shape [batch, seq_len]
        seqlen: shape [batch], 每个样本的有效长度
    Returns:
        packed_input_ids: shape [sum(seqlen)]
    """
    packed = []
    for i in range(input_ids.shape[0]):
        valid_len = seqlen[i].item()
        packed.append(input_ids[i, :valid_len])
    return torch.cat(packed, dim=0)


def pack_logprobs(logprobs: torch.Tensor, seqlen: torch.Tensor) -> torch.Tensor:
    """
    将 logprobs 按 seqlen 拼接成 packed_logprobs。
    每个样本的有效 logprobs 长度为 seqlen[i]-1。
    处理逻辑：对于每个样本 i，跳过第一个元素，取 logprobs[i, 1:seqlen[i]]
    Args:
        logprobs: shape [batch, seq_len]
        seqlen: shape [batch], 每个样本的有效长度
    Returns:
        packed_logprobs: shape [sum(seqlen-1)]
    """
    packed = []
    for i in range(logprobs.shape[0]):
        # 跳过第一个元素，取 seqlen[i]-1 个元素
        packed.append(logprobs[i, 1 : seqlen[i].item()])
    return torch.cat(packed, dim=0)


def pack_ref_logprobs(logprobs: torch.Tensor, seqlen: torch.Tensor) -> torch.Tensor:
    packed = []
    for i in range(logprobs.shape[0]):
        ref_len = seqlen[i].item() - 1
        packed.append(logprobs[i, :ref_len])
    return torch.cat(packed, dim=0)


def pack_prompt_mask(prompt_mask: torch.Tensor, seqlen: torch.Tensor) -> torch.Tensor:
    packed = []
    for i in range(prompt_mask.shape[0]):
        valid_len = seqlen[i].item()
        packed.append(prompt_mask[i, :valid_len])
    return torch.cat(packed, dim=0)


def load_tokenizer(model_path="/sfs_turbo/models/Qwen3_5_35B_A3B/"):
    """加载 tokenizer"""
    tokenizer = AutoTokenizer.from_pretrained(model_path, trust_remote_code=True)
    return tokenizer


def prepare_ppo_data(texts, tokenizer, seq_length=512, prompt_ratio=0.2):
    """
    根据多条输入文本生成 PPO 训练数据集（packed 格式）

    Args:
        texts: 输入文本列表，每条文本包含 prompt + response
        tokenizer: tokenizer 对象
        seq_length: 每条序列的长度（默认 512）
        prompt_ratio: prompt 占总长度的比例（用于生成 ppo_loss_mask）

    Returns:
        包含 inputs (SequenceSample) 和 logprobs 的字典
    """
    batch_input_ids = []
    batch_prompt_masks = []

    for seq_idx, text in enumerate(texts):
        # 1. 编码文本
        input_ids = tokenizer.encode(text, add_special_tokens=False)

        # 2. 截断或填充到指定长度
        if len(input_ids) > seq_length:
            input_ids = input_ids[:seq_length]
        elif len(input_ids) < seq_length:
            pad_id = tokenizer.pad_token_id if tokenizer.pad_token_id is not None else 151643
            input_ids = input_ids + [pad_id] * (seq_length - len(input_ids))

        batch_input_ids.append(input_ids)

        # 3. 生成 prompt_mask (prompt 部分为 True，response 部分为 False)
        train_length = seq_length - 1
        prompt_len = int(train_length * prompt_ratio)
        prompt_mask = [True] * prompt_len + [False] * (train_length - prompt_len)
        batch_prompt_masks.append(prompt_mask)

    # 转为 tensor
    input_ids_tensor = torch.tensor(batch_input_ids, dtype=torch.int64)
    prompt_mask_tensor = torch.tensor(batch_prompt_masks, dtype=torch.bool)
    seqlen_tensor = torch.tensor([seq_length] * len(texts), dtype=torch.int64)

    # 使用 pack 函数进行 packed
    packed_input_ids = pack_input_ids(input_ids_tensor, seqlen_tensor)
    packed_prompt_mask = pack_prompt_mask(prompt_mask_tensor, seqlen_tensor)

    # train data 长度是 seq_length - 1
    train_length = seq_length - 1

    # 生成对应的训练数据（这里用 0 填充，实际使用时替换为真实值）
    # logprobs 需要特殊处理：shape 是 [batch, seq_length-1]
    logprobs_tensor = torch.zeros(len(texts), train_length, dtype=torch.float32)
    packed_logprobs = pack_ref_logprobs(logprobs_tensor, seqlen_tensor)

    old_logp = torch.zeros_like(packed_logprobs)
    advantages = torch.zeros_like(packed_logprobs)
    kl_rewards = torch.zeros_like(packed_logprobs)

    # ppo_loss_mask: prompt 部分为 False，response 部分为 True（和 prompt_mask 相反）
    packed_ppo_mask = ~packed_prompt_mask

    # 构造 SequenceSample
    data_dict = {
        'packed_input_ids': packed_input_ids,
        'old_logp': old_logp,
        'advantages': advantages,
        'kl_rewards': kl_rewards,
        'ppo_loss_mask': packed_ppo_mask
    }

    seqlens = {
        'packed_input_ids': [[seq_length]] * len(texts),
        'old_logp': [[train_length]] * len(texts),
        'advantages': [[train_length]] * len(texts),
        'kl_rewards': [[train_length]] * len(texts),
        'ppo_loss_mask': [[train_length]] * len(texts)
    }

    keys = list(data_dict.keys())
    dtypes = {key: data_dict[key].dtype for key in keys}
    trailing_shapes = {key: () for key in keys}

    inputs = SequenceSample(
        keys=keys,
        ids=[str(i) for i in range(len(texts))],
        dtypes=dtypes,
        trailing_shapes=trailing_shapes,
        seqlens=seqlens,
        data=data_dict,
        metadata={}
    )

    return {
        'inputs': inputs,
        'logprobs': packed_logprobs
    }
